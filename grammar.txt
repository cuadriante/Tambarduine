program             : block

block               : main

main                : line

line                : expr
                    | boolean_neg
                    | boolean_to_true
                    | boolean_to_false
                    | print

var_decl            : RESERVED:SET VAR ASSIGN expr SEMICOLON

var_assigment_list  : VAR ASSIGN expression
                    | var_assigment_list VAR ASSIGN expression

boolean_neg         : SET VAR NEG SEMICOLON

expr                : 
                    | arith-expr
                    | BOOL
                    | condition
                    | if-expr
                    | for-loop

if-expr             : IF condition LBRACE expression RBRACE
                    | IF condition LBRACE expression RBRACE ELSE LBRACE expression RBRACE

for-loop            : FOR VAR TO factor STEP NUMBER LBRACE expression RBRACE                

condition           : arith-expression (EQUALS|DIFFERENT|LESSTHAN|MORETHAN|LESSTHANE|MORETHANE) arith-expression
                    | NEGATIVE condition
                
arith-expr          : arith-expression (PLUS|MINUS) term
                    | term

term                : term (TIMES|DIVIDE|WHOLEDIVIDE|MODULE|POWER) factor
                    | factor

factor              :NUMBER
                    | VAR
                    | LPAREN factor RPAREN

print               : PRINT LPAREN params RPAREN SEMICOLON

params              : param
                    | params ASSIGN param

param               : NUMBER
                    | VAR
                    | STRING
